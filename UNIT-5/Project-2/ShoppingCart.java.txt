class ShoppingCart { 
    private PaymentMethod method; 
 
    public void setPaymentMethod(PaymentMethod method) { 
        this.method = method; 
    } 
 
    public void checkout(int amount) { 
        if (method != null) { 
            method.pay(amount); 
        } else { 
            System.out.println("Select a payment method"); 
        } 
    } 
}
