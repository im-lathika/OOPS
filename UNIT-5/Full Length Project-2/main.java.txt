import java.util.Scanner; 
 
public class Main { 
    public static void main(String[] args) { 
        Scanner s = new Scanner(System.in); 
        Hospital hospital = new Hospital(); 
        int choice; 
 
        do { 
            System.out.println("\nMenu"); 
            System.out.println("1 Add Doctor"); 
            System.out.println("2 Add Nurse"); 
            System.out.println("3 Display Staff"); 
            System.out.println("4 Exit"); 
            System.out.print("Enter choice: "); 
            choice = s.nextInt(); 
            s.nextLine(); 
 
            switch (choice) { 
                case 1: 
                    System.out.print("Enter Name: "); 
                    String dname = s.nextLine(); 
                    System.out.print("Enter ID: "); 
                    int did = s.nextInt(); 
                    System.out.print("Enter Salary: "); 
                    double dsalary = s.nextDouble(); 
                    s.nextLine(); 
                    System.out.print("Enter Specialization: "); 
                    String spec = s.nextLine(); 
 
                    hospital.addStaff(new Doctor(dname, did, dsalary, spec)); 
                    break; 
 
                case 2: 
                    System.out.print("Enter Name: "); 
                    String nname = s.nextLine(); 
                    System.out.print("Enter ID: "); 
                    int nid = s.nextInt(); 
                    System.out.print("Enter Salary: "); 
                    double nsalary = s.nextDouble(); 
                    s.nextLine(); 
                    System.out.print("Enter Shift: "); 
                    String shift = s.nextLine(); 
 
                    hospital.addStaff(new Nurse(nname, nid, nsalary, shift)); 
                    break; 
 
                case 3: 
                    hospital.displayAllStaff(); 
                    break; 
 
                case 4: 
                    System.out.println("Exit"); 
                    break; 
 
                default: 
System.out.println("Invalid choice"); 
} 
} while (choice != 4); 
s.close(); 
} 
}
